# Steps 步骤条

对复杂任务流进行拆分，引导用户流程完成任务。

# 基础用法

0
第一步
0
第二步
0
第三步
显示代码
  <div class="steps">
    <gm-steps :active="active" finish-status="success">
      <gm-step title="第一步"></gm-step>
      <gm-step title="第二步"></gm-step>
      <gm-step title="第三步"></gm-step>
    </gm-steps>
    <el-button type="primary" style="margin-top: 30px" @click="next">下一步</el-button>
  </div>
  <script>
  import { ref } from 'vue'
  export default {
    setup (props, context) {
      let active = ref(0)

      const next= () => {
        if (active.value++ > 2) active.value = 0
      }

      return {
        active,
        next
      }
    }
  }
  </script>
  <style lang='scss' scoped>
  </style>
