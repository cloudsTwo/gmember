显示代码
  <span>显示总页数</span>
  <gm-pagination layout="total, prev, pager, next" :total="100" />

  <span>可以切换每页显示条数</span>
  <gm-pagination :total="1000" 
    :page-sizes="[100, 200, 300, 400]"
    :page-size="100" 
    layout="sizes, prev, pager, next" />

  <span>可以直接跳转到某一页</span>
  <gm-pagination :page-size="100"
      layout="prev, pager, next, jumper"
      :total="1000" />

  <span>完整功能组合</span>
  <gm-pagination :page-sizes="[100, 200, 300, 400]"
      :page-size="100"
      layout="total, sizes, prev, pager, next, jumper"
      :total="1000"
